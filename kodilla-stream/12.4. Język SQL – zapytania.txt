SELECT SUMMARY, NAME
FROM ISSUES JOIN ISSUESLISTS;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY
FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(I.ID) AS NUM_TASKS
FROM USERS U
         JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME;
